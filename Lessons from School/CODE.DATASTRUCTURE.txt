// Programme for infix to Prefix.
#include<stdio.h>
#include<string.h>
#include<ctype.h>
char stack[100];
int top = -1;
void push(char x)
{
    stack[++top] = x;
}
char pop()
{
    if(top == -1)
        return -1;
    else
        return stack[top--];
}
int priority(char x)
{
    if(x == '(')
        return 0;
    if(x == '+' || x == '-')
        return 1;
    if(x == '*' || x == '/')
        return 2;
    return 0;
}
void infixToPrefix(char *exp)
{
    int len = strlen(exp);
    char prefixExp[100];
    int prefixIndex = 0;
    
    for (int i = len - 1; i >= 0; i--)
    {
        if(isalnum(exp[i]))
            prefixExp[prefixIndex++] = exp[i];
        else if(exp[i] == ')')
            push(exp[i]);
        else if(exp[i] == '(')
        {
            while (top != -1 && stack[top] != ')')
                prefixExp[prefixIndex++] = pop();
            if (top != -1)
                pop(); // Pop the remaining ')'
        }
        else
        {
            while (top != -1 && priority(stack[top]) > priority(exp[i]))
                prefixExp[prefixIndex++] = pop();
            push(exp[i]);
        }
    }
    while (top != -1)
        prefixExp[prefixIndex++] = pop();
    
    // Reverse the prefix expression
    for (int i = 0; i < prefixIndex / 2; i++)
    {
        char temp = prefixExp[i];
        prefixExp[i] = prefixExp[prefixIndex - 1 - i];
        prefixExp[prefixIndex - 1 - i] = temp;
    }
    
    prefixExp[prefixIndex] = '\0';
    
    printf("Prefix expression: %s\n", prefixExp);
}
int main()
{
    char exp[100];
    printf("Enter the expression : ");
    scanf("%s",exp);
    printf("\n");
    infixToPrefix(exp);
    
    return 0;
}

int reverseList(struct node **stNode)
{
	struct node *prev = NULL;
	struct node *current = *stNode;
	struct node *next = NULL;

	{
		next = current->next;
		current->next = prev;
		prev = current;
		current = next;	
	}	
	*stNode = prev;
	return 0;
}

int push(int maxsize)
{
	int item;
	if(top==maxsize-1)
	printf("The Stack is in Overflow Condition, can't enter data.\n");
	else
	{
		printf("Enter Element: ");
		scanf("%d",&item);
		top++;
		stack[top]=item;
		printf("%d element entered in the Stack\n",stack[top]);		
	}	
}
int pop()
{
	if(top==-1)
	printf("The Stack is in Underflow Condition, can't delete data.\n");
	else
	{
		printf("%d element deleted from the Stack\n",stack[top]);
		top--;		
	}		
}
int peek()
{
	if(top==-1)
	printf("The Stack is in Underflow Condition, no element is there to show\n");
	else
	{
		printf("%d element is the peek element in the Stack\n",stack[top]);	
	}	
}
int search()
{
	int key,i,flag;
	if(top==-1)
	printf("The Stack is in Underflow Condition, no element is there to search\n");
	else
	{
		printf("Enter the key element: ");
		scanf("%d",&key);
		for(i=top;i>=0;--i)
		{
			if(key==stack[i])
			{
				flag=0;
			}
			else
			flag = 1;
		}
		if(flag==1)
		printf("Element is not present in the stack\n");
		else
		printf("Element is found in the stack\n");	
	}
}
int display()
{
	int i;
	if(top==-1)
	printf("The Stack is in Underflow Condition, no element is there to search\n");
	else
	{
		printf("Elements in the Stack is: ");
		for(i=top;i>=0;--i)
		{
			printf("%d ",stack[i]);
		}
		printf("\n");	
	}	
}

#include<stdio.h>

void deletion(void);

int main(){
	front = -1;
	rear = -1;
	printf("Enter the size of the queue:");
	scanf("%d",&n);
	printf("\n\t Presh\n\t 1 for insertion\n\t 2 for deletion\n\t 3 for display\n\t 4 for Exit\n\t");
	do{
		printf("\nEnter the choice:");
		scanf("%d",&choice);
		switch(choice){
			case 1:{
				insertion();
				break;
			}
			case 2:{
				deletion();
				break;
			}
			case 3:{
				display();
				break;
			}
			case 4:{
				printf("\n\tEXIT POINT");
				break;
			}
			default:{
				printf("\nPlease enter the valid choice(1/2/3/4)\n");
			}
		}
	}
	while(choice!=4);
	return 0;
}
void insertion(){
	if(rear >= n-1){
		printf("\n\tQueue is overflow.....\n");
	}
	else{
		if(front == -1)
		front = 0;
		printf("Enter the item you want to insert:");
		scanf("%d",&item);
		rear++;
		queue[rear]=item;
	}
}
void deletion(){
	if(front==-1 && rear==-1){
		printf("\n\t Queue is underflow\n");
	}
	else{
		printf("Deleted element is:%d\n",queue[front]);
		front++;
		if(front==rear){
			front=-1;
			rear=-1;
		}
	}
}
void display(){
	if(front>=0){
		printf("\nElements in this Queue are:\n");
		for(i=front;i<=rear;i++){
			printf("%d\n",queue[i]);
		}
	}
	else{
		printf("No element present for display");
	}
}

//Code for Doubly Linked List.
#include<stdio.h>
#include<stdlib.h>
#include<malloc.h> 
struct node {
    int data;
    struct node *LeftLink;
    struct node *RightLink;
}*stnode, *ennode;
int size=0;
int getCurrSize(struct node *);
int DlListcreation(int n);
int Insert_inthe_BeginningOfDLL(int);
int Insert_at_AnypositionOfDLL(int,int);
int Insert_inthe_EndOfDLL(int);
int Delete_from_Beginning();
int Deletion_at_specificPosition(int);
int Delete_from_End();
int Free_DoublyLinkedList();
int displayDlList();
int getCurrSize(struct node *node)
{
	int size = 0;
	while(node != NULL)
	{
		node = node->RightLink;
		size++;
	}
	return size;
}
int DlListcreation(int n)
{
    int i, num;
    struct node *fnNode;
    if(n >= 1)
    {
        stnode = (struct node *)malloc(sizeof(struct node));
        if(stnode != NULL)
        {
            printf(" Input data for node 1 : ");
            scanf("%d", &num);
            stnode->data = num;
            stnode->LeftLink = NULL;
            stnode->RightLink = NULL;
            ennode = stnode;
            for(i=2; i<=n; i++)
            {
                fnNode = (struct node *)malloc(sizeof(struct node));
                if(fnNode != NULL)
                {
                    printf(" Input data for node %d : ", i);
                    scanf("%d", &num);
                    fnNode->data = num;
                    fnNode->LeftLink = ennode;
                    fnNode->RightLink = NULL;
                    ennode->RightLink = fnNode;
                    ennode = fnNode;
                }
                else
                {
                    printf(" Memory can not be allocated.");
                    break;
                }
            }
        }
        else
        {
            printf(" Memory can not be allocated.");
            return 0;
        }
    }
}
int Insert_inthe_BeginningOfDLL(int item) 
{
    struct node* newNode = (struct node*)malloc(sizeof(struct node));
    if(newNode != NULL) 
	{
        newNode->data = item;
        newNode->LeftLink = NULL;
        newNode->RightLink = stnode;
        if(stnode != NULL) 
		{
            stnode->LeftLink = newNode;
        }
        stnode = newNode;
        printf("%d element inserted in the Beginning.\n",newNode->data);
        size++;
    } 
	else 
	{
        printf("Sorry, can't enter any data as memory is not allocated\n");
        return 0;
    }
}
int Insert_at_AnypositionOfDLL(int item, int position) {
    int i;
    struct node *ptr1 = stnode, *ptr2 = NULL;
    int currSize = getCurrSize(stnode);
    struct node *newNode = (struct node*)malloc(sizeof(struct node));
    if(newNode != NULL) 
	{
        newNode->data = item;
        newNode->LeftLink = NULL;
        newNode->RightLink = NULL;
    } else 
	{
        printf("Sorry, can't enter any data as memory is not allocated\n");
        return 0;
    }
    if (position < 1 || position > currSize + 1) 
	{
        printf("Wrong Position Entered(Operation Failed), Please enter correct position\n");
        return 0;
    } 
	else if(position == 1) 
	{
        Insert_inthe_BeginningOfDLL(item);
    } 
	else if(position == currSize + 1)
	{
        Insert_inthe_EndOfDLL(item);
    } 
	else 
	{
        for(i = 1; i < position; i++) 
		{
            ptr2 = ptr1;
            ptr1 = ptr1->RightLink;
        }
        if(ptr1 == NULL) 
		{
            printf("Wrong Position entered, so data insertion is not possible.\n");
            return 0;
        } 
		else 
		{
            ptr2->RightLink = newNode;
            newNode->LeftLink = ptr2;
            newNode->RightLink = ptr1;
            ptr1->LeftLink = newNode;
        }
        printf("%d element inserted in the given position of %d\n", newNode->data, position);
        size++;
    }
}
int Insert_inthe_EndOfDLL(int item)
{
	struct node* newNode = (struct node*)malloc(sizeof(struct node));
	if(newNode!=NULL)
	{
		newNode->data=item;
		newNode->LeftLink = ennode;
        newNode->RightLink = NULL;
        ennode->RightLink = newNode;
        ennode = newNode;
        printf("%d element inserted in the End.\n",newNode->data);
        size++;
	}
	else
	{
		printf("Sorry, can't enter any data as memory is not allocated\n");
		return 0;
	}
}
int Delete_from_Beginning()
{
	struct node *ptr;
	ptr=stnode;
	if(ptr == NULL)
	{
		printf("The memory is not allocated, so no deletion will occur.\n");
		return 0;
	}
	else
	{
		printf("%d element is deleted form te beginning.\n",ptr->data);
		stnode=stnode->RightLink;
		free(ptr);
		size--;
	}
}
int Deletion_at_specificPosition(int position) 
{
    if (stnode == NULL) 
	{
        printf("The list is empty. Deletion is not possible.\n");
        return 0;
    }
    struct node* ptr1 = stnode, *ptr2 = NULL;
    int currSize = getCurrSize(stnode);
    if (position < 1 || position > currSize) 
	{
        printf("Wrong position entered (Operation Failed), can't delete data.\n");
        return 0;
    } 
	else if (position == 1) 
	{
        Delete_from_Beginning();
    } 
	else if (position == currSize) 
	{
        Delete_from_End();
    } 
	else 
	{
        for (int i = 1; i < position; i++) 
		{
            ptr2 = ptr1;
            ptr1 = ptr1->RightLink;
        }
        if (ptr1 == NULL) 
		{
            printf("The memory is not allocated, so no deletion will occur.\n");
            return 0;
        } 
		else 
		{
            ptr2->RightLink = ptr1->RightLink;
            ptr1->RightLink->LeftLink = ptr2;
            free(ptr1);
            size--;
        }
        printf("%d element is deleted from %d position.\n", ptr1->data, position);
    }
}
int Delete_from_End() 
{
    if (stnode == NULL) 
	{
        printf("The list is empty. Deletion is not possible.\n");
        return 0;
    }
	struct node* current = stnode;
    while(current->RightLink != NULL) 
	{
        current = current->RightLink;
    }
    if(stnode == current)
	{
        stnode = NULL;
    } 
	else 
	{
        current->LeftLink->RightLink = NULL; 
	}
	printf("%d element is deleted form te End.\n",current->data);
    free(current);
    size--;
}
int Free_DoublyLinkedList() 
{
    struct node* current = stnode;
    while(current != NULL) 
	{
        struct node* temp = current;
        current = current->RightLink;
        free(temp);
    }
    stnode = NULL;
    size = 0; 
}

//Bubble Sort
#include<stdio.h>
#include<stdbool.h>
int swap(int*,int*);
int opt_bubbleSort(int[],int);
int display(int[],int);
int swap(int*x,int*y)
{
	int temp = *x;
		*x = *y;
		*y = temp;
}
int opt_bubbleSort(int arr[],int n)
{
	int i,j;
	bool flag;
	for(i=0;i<n-1;i++)
	{
		flag = false;
		for(j=0;j<n-i-1;j++)
		{
			if(arr[j]>arr[j+1])
			{
				swap(&arr[j],&arr[j+1]);
				flag = true;
			}
		}
		if(flag == false)
		break;
	}
}
int display(int arr[],int size)
{
	int i;
	printf("[ ");
	for(i=0;i<size;i++)
	{
		printf("%d ",arr[i]);
	}
	printf("]");
}

	}
	printf("\nArray before Bubble sorting: ");

}

//Insertion Sort
#include<stdio.h>

int opt_insertionSort(int arr[],int n)
{
	int i,j,key;
	for(j=1;j<n;j++)
	{
		key=arr[j];
		i=j-1;
		while((i>-1)&&arr[i]>key)
		{
			arr[i+1]=arr[i];
			i--;
		}
		arr[i+1]=key;
	}
}

int swap(int*x,int*y)
{
	int temp = *x;
		*x = *y;
		*y = temp;
}
int opt_selectionSort(int arr[],int n)
{
	int i,j,min;
	for(i=0;i<n-1;i++)
	{
		int min=i;
		for(j=i+1;j<n;j++)
		{
			if(arr[j]<arr[min])
			min=j;
		}
		swap(&arr[min],&arr[i]);
	}
}

//Radix Sort
#include<stdio.h>
int getMax(int[],int);
int countSort(int[],int,int);
int opt_radixSort(int[],int);
int display(int[],int);
int getMax(int arr[], int n)
{
    int max=arr[0];
    for(int i=1;i<n;i++) 
	{
        if (arr[i]>max)
		{
            max=arr[i];
        }
    }
    return max;
}
int countSort(int arr[], int n, int exp)
{
    const int BASE = 10;
    int output[n];
    int count[BASE] = {0};
    for(int i=0;i<n;i++) 
	{
        count[(arr[i]/exp)%BASE]++;
    }
    for(int i=1;i<BASE;i++)
	{
        count[i]+=count[i-1];
    }
    for(int i=n-1;i>=0;i--)
	{
        output[count[(arr[i]/exp)%BASE]-1]=arr[i];
        count[(arr[i]/exp)%BASE]--;
    }
    for(int i = 0; i < n; i++)
	{
        arr[i]=output[i];
    }
}
int opt_radixSort(int arr[],int n)
{
	int max=getMax(arr,n);
    for (int exp=1;max/exp>0;exp*=10)
	{
        countSort(arr,n,exp);
    }
}
