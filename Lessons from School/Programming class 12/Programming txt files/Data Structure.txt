{

case 1:
push();

case 2:
pop();
break;
case 3:
show ();
break;
case 4:
oddeven()
break;
case 5:
count()
break;
default:
printf("Press enter to exit");
}

while(choice!=6);
}

void push()
{
int val;
if(top==n-1)
printf("over flow\n");
else
{
printf("Enter the value\n");
scanf("%d",&val);
top++;
stack[top]=val;
}
}

void pop()
{
if(top==-1)
printf("Under flow\n");
else
{
printf("Element removed %d\n"stack[top]);
top--;
}
}

void show()
{
printf ("Stack elements---------------->\n");
if(top==-1)
printf("Stack is empty");
else
{
for(i=top;i>=0;i--)
{
