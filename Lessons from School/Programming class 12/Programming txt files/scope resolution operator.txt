// wap a programme to check anumber perfect or not.(Scope Resolution)
#include<iostream>
using namespace std;
class perfect{
private: int n,s,i;
public:
void acc();
void cal();
void display();
};
void perfect::acc()
{
cout<<"Enter a number";
cin>>n;
}
void perfect::cal()
{
s=0;
for(i=1;i<n;i++)
{
if(n%i==0)
{
s=s+i;
}
}
}
void perfect::display()
{
if(s==n)
cout<<"Perfect";
else
cout<<"Not Perfect";
}
int main()
{
perfect obj;
obj.acc();
obj.cal();
obj.display();
}

//wap a programme to check anumber Krishnamurti or not.
#include<iostream>
using namespace std;
class speci{
private: int n,i,j,s,d,t,f;
public:
void acc();
void cal();
void display();
};
void speci::acc()
{
cout<<"Enter a number";
cin>>n;
t=n;
}
void speci::cal()
{
s=0;
while(n>0)
{
d=n%10;
f=1;
for(j=1;j<=d;j++)
{
f=f*j;
}
s=s+f;
n=n/10;
}
}
void speci::display()
{
if(s==t)
cout<<"Special number";
else
cout<<"Not a special one";
}
int main()
{
speci obj;
obj.acc();
obj.cal();
obj.display();
}

//wap a programme to check a number prime. (using consturctor)(defalut construtor)

prime()
{
c=0;

{
cout<<"enter a number";

if(n%i==0)

// function overloading.

{
return num1+num2;

{
return num1+num2+num3;

};
int main()
{
addition obj;

{
cout<<"Enter a number";

i=1;
}
void cal()
{
while(i<=n)
{
c=0;
for(i=1;i<=n;i++)
{
if(n%i==0)

}

}
};
