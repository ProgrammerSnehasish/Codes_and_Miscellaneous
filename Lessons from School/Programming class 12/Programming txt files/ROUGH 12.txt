//binary search
#include<stdio.h>
int main()
{
int m[100],k=0,p=0,ns,lb=0,i,n;
printf("Enter no of elements");
scanf("%d",&n);
int ub=n-1;
for(i=0;i<n;i++)
{
scanf("%d",&m[i]);
}
printf("enter a number to search\n");
scanf("%d",&ns);
while(lb<=ub)
{
p=(lb+ub)/2;
if(m[p]<ns)
lb=p+1;
if(m[p]>ns)
ub=p-1;
if(m[p]==ns)
{
k=1;
break;
}
}
if(k==1)
printf("Element found");
else
printf("Not found");
}
