#include<stdio.h>
int main()
{
	int i,j,k,n;
	printf("Enter the range");
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		for(j=n-1;j>=1;j=j-2)
		{
			printf("1");
		}
		for(k=1;k<=i;k++)
		{
		printf("%d",i);
		}
		printf("\n");
	}
}

	    	printf(" ");
	    	for(c=1;c<=(2*r)-1;c++)
	    	{
	    		printf("* ");
			}
			printf("\n");
		}
	}
