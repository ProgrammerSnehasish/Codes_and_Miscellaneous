//accept three digits from user between one to nine.display the greatest and smallest three digit number.
#include<stdio.h>
int main()
{
	int a,b,c,gn=0,sn=0;
	printf("enter digits");
	scanf("%d%d%d",&a,&b,&c);
	if(a>b && a>c)
	{
		if(b>c)
		{
			gn=100*a+b*10+c;
			sn=100*b+10*c+a;
		}
	}
	
	else if(b>a && b>c)
	{
		if(a>c)
		{
			gn=b*100+a*10+c;
			sn=c*100+a*10+b;
		}
		else
		{
			gn=100*b+c*10+a;
			sn=a*100+c*10+b;
		}
	}
	else if(c>a && c>b)
	{
		if(a>b)
		{
			gn=c*100+b*10+a;
			sn=a*100+b*10+c;
		}
		}
		printf("max number=%d\nminimum number=%d",gn,sn);	
}

//WAP using recursive fuction to determine the nth term of Fibonacci series.
#include<stdio.h>    
void printFibo(int n)
{    
    static int n1=0,n2=1,n3;    
    if(n>0)
	{    
        n3 = n1 + n2;    
        n1 = n2;    
        n2 = n3;    
        printf("%d ",n3);    
        printFibo(n-1);    
    }    
}    
int main()
{    
    int n;    
    printf("Enter the range you want to print: ");    
    scanf("%d",&n);    
    printf("The Fibonacci Series upto %d term is: ",n);    
    printf("%d %d ",0,1);    
    printFibo(n-2);    
    return 0;  
}

int fact(int n)
{
if(n==0||n==1)
return 1;
else
return n*fact(n-1);
}

	scanf("%d",&len);
	printf("Enter elements for array 1\n");
	for(i=0;i<len;i++){
		for(j=0;j<len;j++){

		}
	}
	printf("The first 2D array of (%dx%d) is:\n",len,len);
	for(i=0;i<len;i++){

			printf("%d",a[i][j]);
		}
		printf("\n");
	}

	for(i=0;i<len;i++){
		for(j=0;j<len;j++){
			printf("Enter element: ");

		}
	}
	printf("The second 2D array of (%dx%d) is:\n",len,len);
	for(i=0;i<len;i++){
		for(j=0;j<len;j++){
			printf("%d",b[i][j]);
		}
		printf("\n");
	}
	int summa=arraysum(a,b,len);
	printf("Summation of two 2D array of (%dx%d) is: \n",len,len);
	for(i=0;i<len;i++){
		for(j=0;j<len;j++){
			sum[i][j]=summa;
		    printf("%d",sum[i][j]);
		}
		printf("\n");
	}
}
int arraysum(int x[100][100],int y[100][100],int l)
{
	int i,j,c[100][100];
	for(i=0;i<l;i++){
		for(j=0;j<l;j++){
			c[i][j]=x[i][j]+y[i][j];		
			return c[i][j];
	}
}
