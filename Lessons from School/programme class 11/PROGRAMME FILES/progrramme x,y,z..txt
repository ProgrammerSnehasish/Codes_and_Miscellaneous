//sunny number
#include<stdio.h>
int main()
{
	int n,i,c=0;
	printf("enter a number");
	scanf("%d",&n);
	n=n+1;
	i=1;
	while(i<=n)
	{
		if(n%i==0)
		c=c+1;
		i++;
	}
	if(c%2==1)
	printf("sunny number");
	else
	printf("not sunny number");
}

//Amicable Pair Number
#include<stdio.h>
int main()
{
int a,b,x,y,i,s1=0,s2=0;
printf("Enter two numbers");
scanf("%d",&a,&b);
a=x;
b=y;
for(i=1;i<a;i++)
{
if(a%i==0)
s1=s1+i;
}
for(i=1;i<b;i++)
{
if(b%i==0)
s2=s2+i;
}
if(s1==y && s2==x)
printf("It is a Amicable Pair Number");
else
printf("Error");
}
