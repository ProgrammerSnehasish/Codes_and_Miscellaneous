main()
{
	int i,n,s=0;
	printf("enter a range");
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		if(i%2==0)
		s=s-i;
		else
		s=s+i;
	}
	printf("sum of the series=%d",s);
}

#include<stdio.h>
int main()
{
int n,i,j,s=0,d,t,f;
printf("Enter a number");
scanf("%d",&n);
t=n;
while(n>0)
{
d=n%10;
f=1;
for(j=1;j<=d;j++)
{
f=f*j;
}
s=s+f;
n=n/10;
}
if (s==t)
printf("%d is a Krishnamurti number",t);
else
printf("Wrong enter");
}
