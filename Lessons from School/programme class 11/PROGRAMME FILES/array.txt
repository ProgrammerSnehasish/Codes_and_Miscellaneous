//binary search
#include<stdio.h>
int main()
{
int m[100],k=0,p=0,ns,lb=0,i,n;
printf("Enter no of elements");
scanf("%d",&n);
int ub=n-1;
for(i=0;i<n;i++)
{
scanf("%d",&m[i]);
}
printf("enter a number to search\n");
scanf("%d",&ns);
while(lb<=ub)
{
p=(lb+ub)/2;
if(m[p]<ns)
lb=p+1;
if(m[p]>ns)
ub=p-1;
if(m[p]==ns)
{
k=1;
break;
}
}
if(k==1)
printf("Element found");
else
printf("Not found");
}

//selection sort technique.
#include<stdio.h>
int main()
{
int m[10],i,j,t,min;
for(i=0;i<10;i++)
{
scanf("%d",&m[i]);
}
for(i=0;i<9;i++)
{
min=i;
for(j=i+1;j<10;j++)
{
if(m[j]<m[min])
min=j;
}
t=m[i];
m[i]=m[min];
m[min]=t;
}
printf("The sorted array\n");
for(i=0;i<10;i++)
printf("%d ",m[i]);
}
