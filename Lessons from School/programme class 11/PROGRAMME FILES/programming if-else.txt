{

	printf("enter digits");
	scanf("%d%d%d",&a,&b,&c);
	if(a>b && a>c)
	{
		if(b>c)
		{
			gn=100*a+b*10+c;
			sn=100*b+10*c+a;
		}
	}
	
	else if(b>a && b>c)
	{
		if(a>c)
		{
			gn=b*100+a*10+c;
			sn=c*100+a*10+b;
		}
		else
		{
			gn=100*b+c*10+a;
			sn=a*100+c*10+b;
		}
	}
	else if(c>a && c>b)
	{
		if(a>b)
		{

		}
		}
