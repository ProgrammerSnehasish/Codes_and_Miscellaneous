//C Programme to implement Cyclic Redundency Check(CRC)
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#define length_of_generatingPolynomial strlen(generatingPolynomial)
#define maxsize 32
char data[maxsize], checksum[maxsize], generatingPolynomial[maxsize];
int data_length,indexCRC,indexXOR,i,index;
void XOR();
void receiver();
void crc();
void XOR() {
    for (indexXOR = 0; indexXOR < length_of_generatingPolynomial; indexXOR++) {
        checksum[indexXOR] = ((checksum[indexXOR] == generatingPolynomial[indexXOR]) ? '0' : '1');
    }
}
void receiver() {
    crc();
    for(int index = 0; (index < length_of_generatingPolynomial - 1) && (checksum[index] != '1'); index++)
    	if (index < length_of_generatingPolynomial-1)
        	printf("\nError detected");
        else
        	printf("\nNo error detected\n\n");
}
void crc() {
    for (indexCRC = 0; indexCRC < length_of_generatingPolynomial; indexCRC++)
        checksum[indexCRC] = data[indexCRC];
    do {
        if (checksum[0] == '1')
            XOR();
        for (i=0;i<length_of_generatingPolynomial-1;i++)
            checksum[i] = checksum[i + 1];
        checksum[i] = data[indexCRC++];
    } while (indexCRC <= data_length + length_of_generatingPolynomial - 1);
}
int main() {
	int error;
    printf("Enter data to be transmitted: ");
    scanf("%s", data);
    printf("----------------------------------------");
    printf("\nEnter the Generating polynomial: ");
    scanf("%s", generatingPolynomial);
    data_length = strlen(data);
    for (int index = data_length; index < data_length + length_of_generatingPolynomial - 1; index++) {
        data[index] = '0';
    }
    printf("----------------------------------------");
    printf("\n Data padded with n-1 zeros(Modified Data): %s", data);
    printf("\n----------------------------------------");
    crc();
    printf("\nCRC or Check value is : %s", checksum);
    for (int i = data_length; i < data_length + length_of_generatingPolynomial - 1; i++) {
        data[i] = checksum[i - data_length];
    }
    printf("\n----------------------------------------");
    printf("\n Final data to be sent(Final Codeword): %s", data);
    printf("\n----------------------------------------\n");
    printf("Please select your choice for test error detection:\n1. Press 0(Yes) to enter error bit.\n2. Press 1(No).\n");
    printf("Please enter your choice:");
	scanf("%d",&error);
	if(error == 0)
	{
		do{
			printf("Enter the position where error is to be inserted:");
			scanf("%d",&error);
		}while(error == 0 || error > data_length + length_of_generatingPolynomial - 1);
		data[error-1]=(data[error-1]=='0')?'1':'0';
		printf("----------------------------------------\n");
		printf("Erroneous data:%s",data); 
		receiver();
	}
	else if(error==1)
	{
    	printf("-----------------------------\n");
    	printf("Data received: %s", data);
    	printf("\nNo error detected\n");
	}
	else
	printf("Wrong choice entered.");
    return 0;
}

#include<stdio.h>
#include<stdlib.h>
#define max_size 100
int arr[max_size];
int get_array(int[],int);
int display_list(int[],int);
int merge(int[],int,int,int);
int merge_sort(int[],int,int);
int get_array(int arr[max_size],int no_of_elements)
{
	int index;
	for(index=0;index<no_of_elements;index++)
	{
		printf("Enter element %d:",index+1);
		scanf("%d",&arr[index]);
	}
}
int display_list(int arr[max_size],int no_of_element)
{
	int index;
	printf("[\t");
	for(index=0;index<no_of_element;index++)
	{
		printf("%d\t",arr[index]);
	}
	printf("]\n");
}
int merge(int arr[max_size],int left,int mid,int right)
{
	int leftIndex,rightIndex,index;
	int leftlim=mid-left+1;
	int rightlim=right-mid;
	int Left[leftlim],Right[rightlim];
	for(leftIndex=0;leftIndex<leftlim;leftIndex++)
	{
		Left[leftIndex]=arr[left+leftIndex];
	}
	for(rightIndex=0;rightIndex<rightlim;rightIndex++)
	{
		Right[rightIndex]=arr[mid+1+rightIndex];
	}
	leftIndex=0;
	rightIndex=0;
	index=left;
	while(leftIndex<leftlim && rightIndex<rightlim){
		if(Left[leftIndex]<=Right[rightIndex]){
			arr[index]=Left[leftIndex];
			leftIndex++;
		}else{
			arr[index]=Right[rightIndex];
			rightIndex++;
		}
		index++;
	}
	while(leftIndex<leftlim){
		arr[index]=Left[leftIndex];
		leftIndex++;
		index++;
	}
	while(rightIndex<rightlim){
		arr[index]=Right[rightIndex];
		rightIndex++;
		index++;
	}
}
int merge_sort(int arr[max_size],int left,int right)
{
	if(left<right)
	{
		int mid=left+(right-left)/2;
		merge_sort(arr,left,mid);
		merge_sort(arr,mid+1,right);
		merge(arr,left,mid,right);
	}
}
int main()
{
	int no_of_elements;
	printf("_________Merge Sort_________\n");
	printf("Enter total number of elements:");
	scanf("%d",&no_of_elements);
	get_array(arr,no_of_elements);
	printf("The array before sorting:");
	display_list(arr,no_of_elements);
	printf("The arrya after sorting:");
	merge_sort(arr,0,no_of_elements-1);
	display_list(arr,no_of_elements);
}

	{

int swap(int *element1,int *element2)
{
	int temp=*element1;
		*element1=*element2;
		*element2=temp;
}
int partition(int arr[max_size],int low,int high)
{
	int pivot=arr[low];
	int leftIndex=low;
	int rightIndex=high;
	while(leftIndex<rightIndex){
		while(arr[leftIndex]<=pivot && leftIndex<=high-1){
			leftIndex++;
		}
		while(arr[rightIndex]>pivot && rightIndex>=low+1){
			rightIndex--;
		}
		if(leftIndex<rightIndex){
			swap(&arr[leftIndex],&arr[rightIndex]);
		}
	}
	swap(&arr[low],&arr[rightIndex]);
	return rightIndex;
}
int quick_sort(int arr[max_size],int low,int high)
{
	if(low<high){
		int partitionIndex=partition(arr,low,high);
		quick_sort(arr,low,partitionIndex-1);
		quick_sort(arr,partitionIndex+1,high);
	}
}

{

	display_list(arr,no_of_elements);
	printf("The arrya after sorting:");
	quick_sort(arr,0,no_of_elements-1);
	display_list(arr,no_of_elements);
}

int maxHeapify(int arr[max_size], int no_of_elements, int index)
{
    int largest = index;
    int left = 2 * index + 1;
    int right = 2 * index + 2;
    
    if (left < no_of_elements && arr[left] > arr[largest])
        largest = left;
    if (right < no_of_elements && arr[right] > arr[largest])
        largest = right;
    if (largest != index)
    {
        swap(&arr[index], &arr[largest]);
        maxHeapify(arr, no_of_elements, largest);
    }
    return 0;
}
int maxHeap_sort(int arr[max_size], int no_of_elements)
{
    int index;
    for (index = no_of_elements / 2 - 1; index >= 0; index--)
        maxHeapify(arr, no_of_elements, index);
    for (index = no_of_elements - 1; index >= 0; index--)
    {
        swap(&arr[0], &arr[index]);
        maxHeapify(arr, index, 0);
    }
    return 0;
}
int minHeapify(int arr[max_size], int no_of_elements, int index)
{
    int smallest = index;
    int left = 2 * index + 1;
    int right = 2 * index + 2;
    
    if (left < no_of_elements && arr[left] < arr[smallest])
        smallest = left;
    if (right < no_of_elements && arr[right] < arr[smallest])
        smallest = right;
    if (smallest != index)
    {
        swap(&arr[index],&arr[smallest]);
        minHeapify(arr,no_of_elements,smallest);
    }
    return 0;
}
int minHeap_sort(int arr[max_size], int no_of_elements)
{
    int index;
    for(index=no_of_elements/2-1;index>=0;index--)

    for (index=no_of_elements-1;index>=0;index--)
    {
        swap(&arr[0],&arr[index]);
        minHeapify(arr,index,0);
    }
    return 0;
}

//C programme to implement B.F.S and D.F.S.
#include <stdio.h>
#define MAX_SIZE 100
int adj[MAX_SIZE][MAX_SIZE], visited[MAX_SIZE], queue[MAX_SIZE], stack[MAX_SIZE];
int top = -1, front = -1, rear = -1;
void get_AdjacencyMatrix(int no_of_vertices);
void get_operationalChoice(int no_of_vertices);
void add(int item);
int deletion();
void push(int item);
int pop();
void bfs(int src_v, int no_of_vertices);
void dfs(int src_v, int no_of_vertices);
void get_AdjacencyMatrix(int no_of_vertices)
{
    int index, index2;
    for (index = 0; index < no_of_vertices; index++)
    {
        for (index2 = 0; index2 < no_of_vertices; index2++)
        {
            printf("Enter 1 if %d has a node with %d else 0: ", index + 1, index2 + 1);
            scanf("%d", &adj[index][index2]);
        }
    }
    printf("The Adjacency Matrix is:\n");
    for (index = 0; index < no_of_vertices; index++)
    {
        for (index2 = 0; index2 < no_of_vertices; index2++)
        {
            printf("%d\t", adj[index][index2]);
        }
        printf("\n");
    }
}
void add(int item)
{
    if (rear == MAX_SIZE - 1)
    {
        printf("\nQueue is overflow.....\n");
    }
    else
    {
        if (front == -1)
            front = 0;
        rear++;
        queue[rear] = item;
    }
}
int deletion()
{
    if ((front > rear) || (front == -1))
        return 0;
    else
    {
        int k = queue[front++];
        return k;
    }
}
void push(int item)
{
    if (top == MAX_SIZE - 1)
    {
        printf("The Stack is in Overflow Condition, can't enter data.\n");
    }
    else
    {
        top++;
        stack[top] = item;
    }
}
int pop()
{
    if (top == -1)
        return 0;
    else
    {
        int k = stack[top--];
        return k;
    }
}
void bfs(int src_v, int no_of_vertices)
{
    int p, index;
    add(src_v);
    visited[src_v] = 1;
    p = deletion();
    if (p != 0)
        printf(" %d ", p);
    while (p != 0)
    {
        for (index = 0; index < no_of_vertices; index++)
            if ((adj[p][index] != 0) && (visited[index] == 0))
            {
                add(index);
                visited[index] = 1;
            }
        p = deletion();
        if (p != 0)
            printf(" %d ", p);
    }
    for (index = 0; index < no_of_vertices; index++)
        if (visited[index] == 0)
            bfs(index, no_of_vertices);
}
void dfs(int src_v, int no_of_vertices)
{
    int index, k;
    push(src_v);
    visited[src_v] = 1;
    k = pop();
    if (k != 0)
        printf(" %d ", k);
    while (k != 0)
    {
        for (index = 0; index < no_of_vertices; index++)
            if ((adj[k][index] != 0) && (visited[index] == 0))
            {
                push(index);
                visited[index] = 1;
            }
        k = pop();
        if (k != 0)
            printf(" %d ", k);
    }
    for (index = 0; index < no_of_vertices; index++)
        if (visited[index] == 0)
            dfs(index, no_of_vertices);
}

{
    int index, choice, src_vertex;

void fractionalknapsack(int n, float weight[], float profit[], float capacity) 
{
    float x[max_size], tp = 0;
    int i, j;
    float ratio[max_size];
    for (i = 0; i < n; i++) {
        ratio[i] = profit[i] / weight[i];
    }
    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - i - 1; j++) {
            if (ratio[j] < ratio[j + 1]) {
                swap(&ratio[j],&ratio[j+1]);
                swap(&weight[j],&weight[j+1]);
                swap(&profit[j],&profit[j+1]);
            }
        }
    }
    for (i = 0; i < n; i++) {
        x[i] = 0.0;
    }
    for (i = 0; i < n; i++) {
        if (weight[i] <= capacity) {
            x[i] = 1.0;
            tp += profit[i];
            capacity -= weight[i];
        } else {
            x[i] = capacity / weight[i];
            tp += (x[i] * profit[i]);
            break; // We have filled the knapsack to its capacity
        }
	}
    printf("\nThe result of item set (portion choosing):- ");
    display_list(x,n);
    printf("\nMaximum profit is:- %f", tp);
}

//N-Queens Problems.
#include<stdio.h>
#include<stdlib.h>
#define MAX_SIZE 100
int position[MAX_SIZE]={0};
int place(int);
int print_chessboard(int);
int queen(int,int);
int place(int present_queen){
	int i;
	for(i=1;i<present_queen;i++){
		if((position[i]==position[present_queen])||((abs(position[i]-position[present_queen])== abs(i-present_queen))))
			return 0;
		}
		return 1;
}
int print_chessboard(int no_of_queens){
	int i,j;
	printf("Final Chess Board after placing all queens:\n");
	for(i=1;i<=no_of_queens;i++){
		for(j=1;j<=no_of_queens;j++){
			if(position[i]==j)
				printf("Q%d\t",i);
			else
				printf("_\t");
		}
		printf("\n");
	}
}
int queen(int present_queen,int no_of_queens){
	position[present_queen]=position[present_queen]+1;
	while(position[present_queen]<=no_of_queens && !place(present_queen)){
		position[present_queen]++;
	}
	if(position[present_queen]<=no_of_queens){
		if(present_queen==no_of_queens)
		{
			print_chessboard(no_of_queens);
			return 1;
		}
	else{
		position[++present_queen]=0;
		queen(present_queen,no_of_queens);
	}
	}
	else{
		queen(present_queen-1,no_of_queens);
	}
}
int main()
{
		int no_of_queens;
		printf("_________N Queen Problem_________\n");
		printf("Enter total number of queens: ");
		scanf("%d",&no_of_queens);
		queen(1,no_of_queens);
}

//C Programme to implement Prim's Algorithm.
#include<stdio.h>
#define max_vertices 100
#define infinity 999
int cost[max_vertices][max_vertices],visited[max_vertices]={0},mincost=0;
int get_costMatrix(int);
int display_costMatrix(int);
void primalgo(int);

void primalgo(int number_of_vertices)
{
	int row,column,temp,temp2,temp3,temp4,edge_count=1,min;
	for(row=1;row<=number_of_vertices;row++){
		for(column=1;column<=number_of_vertices;column++){
			if(cost[row][column]==0){
				cost[row][column]=infinity;
			}
		}
	}
	visited[1]=1;
	while(edge_count<number_of_vertices)
	{
		min=infinity;
		for(row=1;row<=number_of_vertices;row++){
			if(visited[row]==1){
				for(column=1;column<=number_of_vertices;column++){
					if(cost[row][column]<min && visited[column]==0){
						min=cost[row][column];
						temp=temp2=row;
						temp3=temp4=column;
					}
				}
			}
		}
		if(min==infinity)
		{
			printf("No Spanning tree exists.\n");
			return;
		}
		printf("\n Edge %d:(%d-%d)cost: %d",edge_count++,temp,temp3,min);
		mincost+=min;
		visited[temp3]=1;
		cost[temp][temp3]=cost[temp3][temp]=infinity;
	}
	printf("\nMinimum Cost is(Total minimum cost of the spanning tree): %d\n",mincost);
}
