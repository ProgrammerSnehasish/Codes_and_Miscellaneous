The task is to find the empty packets(0) of chocolate and push it to the end of the conveyor belt(array).(TCS NQT Problem)*/

#define SIZE 100
static int top1 = -1, top2 = -1, top3 = -1;

void push(int stack[], int *top, int element)
{
    if(*top >= SIZE - 1)
    {
        printf("Stack Overflow\n");
        return;
    }
    stack[++(*top)] = element;
}

{
    int index;
    for(index = 0; index <= top1; index++)

/* The problem statement is “A positive integer has been given as an input. 
Convert decimal value to binary representation. Toggle all bits of it after the most significant bit including the most significant bit. 
Print the positive integer value after toggling all bits”.*/
#include <stdio.h>
#include <math.h>
#define size 32
int decimalToBinary(int);
int toggleBits(int);
int decimalToBinary(int decimal_number){
    int binary[size];
    int i = 0,j;
    if (decimal_number == 0) {
        printf("Binary: 0\n");
        return;
    }
    while (decimal_number > 0){
        binary[i] = decimal_number % 2;
        decimal_number = decimal_number / 2;
        i++;
    }
    printf("Binary: ");
    for (j = i - 1; j >= 0; j--) {
        printf("%d", binary[j]);
    }
    printf("\n");
}
int toggleBits(int decimal_number) {
    int numBits = (int)log2(decimal_number) + 1;
    int mask = (1 << numBits) - 1;
    int toggledNumber = decimal_number ^ mask;
    return toggledNumber;
}
int main(){
    int decimal_number;
    printf("Enter an integer: ");
    scanf("%d", &decimal_number);
    if (decimal_number >= 1 && decimal_number <= 100){
    	printf("The binary number of the given decimal number is: ");
    	decimalToBinary(decimal_number);
        int result = toggleBits(decimal_number);
        printf("The toggled binary digit is: ");
        decimalToBinary(result);
        printf("The positive integer after toggling is(converting toggled binary bits to decimal):%d\n", result);
    } else {
        printf("Input out of range. Please enter a number between 1 and 100.\n");
    }
    
    return 0;
}

void swap(int*x,int*y)
{
	int temp = *x;
		*x = *y;

}

//Find majority element in the array.(Logic: the element which appears more than n/2 times.)
#include <stdio.h>
#define MAX_SIZE 100
int get_value();
void get_array(int [], int);
void displayList(int [], int);
int findMajorelement(int [], int);
int get_value() {
    int value;
    scanf("%d", &value);
    return value;
}
void get_array(int arr[MAX_SIZE], int totalNumber_ofElements) {
    int index;
    for (index = 0; index < totalNumber_ofElements; index++) {
        printf("Enter element for index %d: ", index);
        scanf("%d", &arr[index]);
    }
}
void displayList(int arr[MAX_SIZE], int totalNumber_ofElements) {
    int index;
    printf("[ ");
    for (index = 0; index < totalNumber_ofElements; index++) {
        printf("%d ", arr[index]);
    }
    printf("]\n");
}
int findMajorelement(int arr[MAX_SIZE], int totalNumber_ofElements) {
    int index, major_element_candidate = arr[0], count = 1;
    for (index = 1; index < totalNumber_ofElements; index++) {
        if (arr[index] == major_element_candidate) {
            count++;
        } else {
            count--;
        }
        if (count == 0) {
            major_element_candidate = arr[index];
            count = 1;
        }
    }
    count = 0;
    for (index = 0; index < totalNumber_ofElements; index++) {
        if (arr[index] == major_element_candidate) {
            count++;
        }
    }
    
    if (count > totalNumber_ofElements / 2) {
        return major_element_candidate;
    } else {
        return -1;
    }
}
int main() {
    int totalNumber_ofElements, arr[MAX_SIZE];
    printf("Enter total number of elements: ");
    totalNumber_ofElements = get_value();
    printf("Enter values for array:\n");
    get_array(arr, totalNumber_ofElements);
    printf("The array is: ");
    displayList(arr, totalNumber_ofElements);
    int result = findMajorelement(arr, totalNumber_ofElements);
    if (result != -1) {
        printf("The majority element in the array is: %d\n", result);
    } else {
        printf("No majority element found in the array.\n");
    }
    return 0;
}

//Find row which has maximum number of 1's in a 2D matrix.
#include <stdio.h>
#define MAX_SIZE 100
int arr[MAX_SIZE][MAX_SIZE];
void get_Matrix(int);
void display_Matrix(int);
void findRowHavingMaxOnes(int);
void get_Matrix(int dimension) {
    int row, column;
    printf("Enter data for matrix (only in the form of 0's and 1's)...\n");
    for (row = 0; row < dimension; row++) {
        for (column = 0; column < dimension; column++) {
            printf("Enter value for [%d][%d]: ", row, column);
            scanf("%d", &arr[row][column]);
        }
    }
}
void display_Matrix(int dimension) {
    int row, column;
    printf("Displaying Matrix...\n");
    for (row = 0; row < dimension; row++) {
        for (column = 0; column < dimension; column++) {
            printf("%d\t", arr[row][column]);
        }
        printf("\n");
    }
}
void findRowHavingMaxOnes(int dimension) {
    int row, column;
    int maxOnes = -1, maxRowIndex = -1;
    for (row = 0; row < dimension; row++) {
        int countOnes = 0;
        for (column = 0; column < dimension; column++) {
            if (arr[row][column] == 1) {
                countOnes++;
            }
        }
        if (countOnes > maxOnes) {
            maxOnes = countOnes;
            maxRowIndex = row;
        }
    }
    if (maxRowIndex != -1) {
        printf("Row %d has the maximum number of 1's: %d\n", maxRowIndex, maxOnes);
    } else {
        printf("No 1's found in the matrix.\n");
    }
}
int main() {
    int dimension;
    printf("Enter the dimension of the matrix: ");
    scanf("%d", &dimension);
    get_Matrix(dimension);
    display_Matrix(dimension);
    findRowHavingMaxOnes(dimension);
    return 0;
}

    }

	int result=1;
	base=base%mod;
	while(exp>0)
	{
		if(exp%2==1)
			result=(result*base)%mod;
		exp=exp>>1;	
		base=(base*base)%mod;
	}
	return result;
}

        return '0' + (mod - 53);
    } else if (mod >= 63 && mod <= 78) {
        return 'i' + (mod - 63);
    }
    return '?';
}

int isPrime(int num) {
	int i;
    if (num <= 1) return 0;
    if (num <= 3) return 1;
    if (num % 2 == 0 || num % 3 == 0) return 0;
    for (i = 5; i * i <= num; i += 6) {
        if (num % i == 0 || num % (i + 2) == 0) return 0;
    }
    return 1;
}

	scanf("%d",&k);
	int private_key = generate_privateKey(private_key_generatingFactor,k,e);
	getchar();
	printf("The private key is: %d\n",private_key);
	printf("Enter a message to encrypt: ");

    	size_t len = strlen(input);
	if (len > 0 && input[len - 1] == '\n') {
		input[len - 1] = '\0';
	}
	convertToNumberedString(input, output);
	printf("Numbered string(given message): %s\n", output);
