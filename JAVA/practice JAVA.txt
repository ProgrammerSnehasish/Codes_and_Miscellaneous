    static Scanner scanner = new Scanner(System.in);
    public static void get_list(int list[], int no_of_elements) {
        System.out.println("Please enter elements in sorted order(descending order):");
        for (int index = 0; index < no_of_elements; index++) {
            System.out.print("Enter element " + (index + 1) + ": ");
            list[index] = scanner.nextInt();
        }
    }
    public static void display_list(int[] list,int no_of_elements){
        System.out.print("[\t");
        for(int index = 0;index < no_of_elements;index++){
            System.out.print(list[index]+"\t");
        }
        System.err.print("]\n");
    }
    public static int binarySearch(int[] list, int low, int high, int key_element) {
        while(low<=high){
            int mid = low + (high - low)/2;
            if(list[mid]==key_element)
                return mid;// if element found.
            if(list[mid]<key_element)
                low = mid + 1;
            else
                high = mid - 1;
        }
        return -1; // Return -1 if the element is not found.
    }
    public static void main(String[] args) {
        System.out.println("________Binary Search________");
        System.out.print("Enter total number of elements: ");
        int no_of_elements = scanner.nextInt();
        int[] list = new int[no_of_elements];
        get_list(list, no_of_elements);
        System.out.print("The List is:");
        display_list(list, no_of_elements);
        System.out.print("Enter key element:");
        int key_element = scanner.nextInt();
        int result = binarySearch(list,0,no_of_elements-1,key_element);
        if (result != -1) {
            System.out.println("Element found at index: " + result);

            System.out.println("Element not found.");

        arr[j] = temp;
    }
    public static void get_list(int list[], int no_of_elements) {
        for (int index = 0; index < no_of_elements; index++) {
            System.out.print("Enter element " + (index + 1) + ": ");

            System.out.print(list[index]+"\t");
        }
        System.err.print("]\n");
    }
    public static void bubbleSort(int[] list, int no_of_elements) {
        int index,index2;
	    int flag;
	    for(index=0;index<no_of_elements-1;index++){
            flag = 0;
            for(index2=0;index2<no_of_elements-index-1;index2++){
                if(list[index2]>list[index2+1])
                {
                    swap(list,index2,index2+1);
                    flag = 1;
                }
            }
            if(flag == 0)
            break;
        }
    }
    public static void main(String[] args) {
        System.out.println("________Bubble Sort________");

        if (present_queen > no_of_queens) {
            print_chessboard(no_of_queens);
            return true;
        } else {
            for (int i = 1; i <= no_of_queens; i++) {
                position[present_queen] = i;
                if (place(present_queen)) {
                    if (queen(present_queen + 1, no_of_queens)) {
                        return true;
                    }
                }
            }
        }
        return false;
    }
    public static boolean place(int present_queen) {
        for (int index = 1; index < present_queen; index++) {
            if ((position[index] == position[present_queen]) || 
                (Math.abs(position[index] - position[present_queen]) == Math.abs(index - present_queen))) {
                return false;
            }
        }
        return true;
    }
    public static void print_chessboard(int no_of_queens) {
        System.out.println("Final Chess Board after placing all queens:\n");

                if (position[row] == column)
                    System.out.print("Q"+row+"\t");
                else
                    System.out.print("_\t");

            System.out.print("\n");
        }
    }

//Prim's Algorithm.(04-07-2024)[Moderate Lv]
import java.util.Scanner;
class PrimAlgorithm {
    private static final int infinity_value = 999;
    static int cost[][] = new int[100][100];
    static int visited[] = new int[100];
    static int mincost = 0;
    static Scanner scanner = new Scanner(System.in);

    public static void get_costMatrix(int number_of_vertices) {
        int row, column;
        System.out.println("Enter data for cost matrix...");
        for (row = 1; row <= number_of_vertices; row++) {
            for (column = 1; column <= number_of_vertices; column++) {
                System.out.print("Enter the path weight value/cost of [" + row + "][" + column + "]: ");
                cost[row][column] = scanner.nextInt();
                if (cost[row][column] == 0 && row != column) {
                    cost[row][column] = infinity_value; // Set infinity_value for no path (except self-loops)
                }
            }
        }
    }
    public static void display_costMatrix(int number_of_vertices) {
        int row, column;
        System.out.println("Displaying Cost Matrix (Adjacency Matrix)...");
        for (row = 1; row <= number_of_vertices; row++) {
            for (column = 1; column <= number_of_vertices; column++) {
                System.out.print(cost[row][column] + "\t");
            }
            System.out.print("\n");
        }
    }
    public static void primalgo(int number_of_vertices) {
        int row, column, temp = 0, temp2 = 0, temp3 = 0, temp4 = 0, edge_count = 1, min;
        visited[1] = 1;

        while (edge_count < number_of_vertices) {
            min = infinity_value;
            for (row = 1; row <= number_of_vertices; row++) {
                if (visited[row] == 1) {
                    for (column = 1; column <= number_of_vertices; column++) {
                        if (cost[row][column] < min && visited[column] == 0) {
                            min = cost[row][column];
                            temp2 = row;
                            temp = temp2;
                            temp4 = column;
                            temp3 = temp4;
                        }
                    }
                }
            }
            if (min == infinity_value) {
                System.out.println("No Spanning Tree exists.");
                return;
            }
            System.out.println("Edge " + edge_count + ": (" + temp + "-" + temp3 + ") cost: " + min);
            mincost += min;
            visited[temp3] = 1;
            cost[temp][temp3] = cost[temp3][temp] = infinity_value;
            edge_count++;
        }
        System.out.println("Minimum spanning tree exists...");
        System.out.println("Toatal minimum cost of the Spanning tree is: " + mincost);
    }
    public static void main(String[] args) {
        System.out.println("________Prim's Algorithm________");
        System.out.print("Enter total number of vertices: ");
        int number_of_vertices = scanner.nextInt();
        get_costMatrix(number_of_vertices);
        display_costMatrix(number_of_vertices);
        primalgo(number_of_vertices);
    }
}

//Fractional Knapsack Problem.(05-07-2024)[Moderate Lv]
import java.util.Scanner;
class FractionalKnapsack {
    static Scanner scanner = new Scanner(System.in);
    public static void swap(float[] arr, int i, int j) {
        float temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
    public static void get_list(float list[], int no_of_items) {
        for (int index = 0; index < no_of_items; index++) {
            System.out.print("Enter for element " + (index + 1) + ": ");
            list[index] = scanner.nextFloat();
        }
    }
    public static void displayList(int no_of_items, float[] profit, float[] weight) {
        System.out.println("Item\tProfit\tWeight\tProfit/Weight Ratio");
        for (int i = 0; i < no_of_items; i++) {
            float ratio = profit[i] / weight[i];
            System.out.println((i + 1) + "\t" + profit[i] + "\t" + weight[i] + "\t" + ratio);
        }
    }
    public static void fractionalKnapsack(int no_of_items, float[] weight, float[] profit, float max_capacity) {
        float[] x = new float[no_of_items];
        float[] ratio = new float[no_of_items];
        for (int i = 0; i < no_of_items; i++) {
            ratio[i] = profit[i] / weight[i];
        }
        // Sorting items by their profit/weight ratio in descending order
        for (int i = 0; i < no_of_items - 1; i++) {
            for (int j = 0; j < no_of_items - i - 1; j++) {
                if (ratio[j] < ratio[j + 1]) {
                    swap(ratio, j, j + 1);
                    swap(profit, j, j + 1);
                    swap(weight, j, j + 1);
                }
            }
        }
        float total_profit = 0;
        for (int i = 0; i < no_of_items; i++) {
            if (weight[i] <= max_capacity) {
                x[i] = 1.0f;
                max_capacity -= weight[i];
                total_profit += profit[i];
            } else {
                x[i] = max_capacity / weight[i];
                total_profit += profit[i] * (max_capacity / weight[i]);
                break;
            }
        }
        System.out.println("The result of item set (portion choosing):");
        System.out.println("Item\tItem set");
        for (int i = 0; i < no_of_items; i++) {
            System.out.println((i + 1) + "\t" + x[i]);
        }
        System.out.println("Maximum profit: " + total_profit);
    }
    public static void main(String[] args) {
        System.out.println("________Fractional Knapsack Problem________");
        System.out.print("Enter total number of elements: ");
        int no_of_items = scanner.nextInt();
        float[] profit = new float[no_of_items];
        float[] weight = new float[no_of_items];
        System.out.println("Enter profit:");
        get_list(profit, no_of_items);
        System.out.println("Enter weight:");
        get_list(weight, no_of_items);
        System.out.println("Showing item list with profit, weight, and profit/weight ratio:");
        displayList(no_of_items, profit, weight);
        System.out.print("Enter maximum capacity of the knapsack: ");
        float max_capacity = scanner.nextFloat();
        fractionalKnapsack(no_of_items, weight, profit, max_capacity);
    }
}

import java.util.Scanner;
class Fibonacci{
	public static void Fibbonacci(int range){
		int num1=0,num2=1;
		System.out.println("The Fibbonacci Series without recursion:");
		for(int i=0;i<range;i++){
			System.out.print(num1 + " ");
			int num3 = num2 + num1;
			num1 = num2;
			num2 = num3;
	}
	System.out.println("\n");
 }
	public static int RecFib(int range){
		if(range == 0)
			return 0;
		else if(range == 1)
			return 1;
		else 
		      return RecFib(range-1)+RecFib(range-2);
 }
	public static void main(String [] args){
		Scanner scanner = new Scanner(System.in);
		System.out.print("Enter your range:");
		int range = scanner.nextInt();
		Fibbonacci(range);
		System.out.println("The Fibbonacci Series using recursion:");
		for(int i=0;i<range;i++){
		System.out.print(RecFib(i)+" ");
	}
 }
}

class Pattern{
	public static void PatternLeftAngeledTriangle(int length){
		System.out.println("The pattern of the Left Angled Triangle of length "+length+" is:");
		for(int i=0;i<length;i++){
			for(int j=0;j<=i;j++){
				System.out.print("*");
		}
		System.out.println("");
	}
 }
	public static void PatternRightAngeledTriangle(int length){
		System.out.println("The pattern of the Right Angled Triangle of length "+length+" is:");
		for(int i=1;i<=length;i++){
			for(int j=length-1;j>=i;j--){
				System.out.print(" ");
		}
		for(int k=1;k<=i;k++){
			System.out.print("*");
		}
		System.out.println("");
   }
 }
	public static void PatternTriangle(int length){
		System.out.println("The pattern of the Triangle of length "+length+" is:");
		for(int i=1;i<=length;i++){
			for(int j=length-1;j>=i;j--){
				System.out.print(" ");
			}
			for(int k=1;k<=2*i-1;k++){
				System.out.print("*");
			}
		System.out.println("");
	}
 }
	public static void main(String [] args){
		Scanner scanner = new Scanner(System.in);

		int length = scanner.nextInt();
		PatternLeftAngeledTriangle(length);
		PatternRightAngeledTriangle(length);
		PatternTriangle(length);
 }

//Java code to check maximum and minimum occuring word in a String.
import java.util.Scanner;
class MaxCharacter {
    public static void main(String[] args) {    
        Scanner scanner = new Scanner(System.in);
        System.out.println("Enter a string: ");
        String str = scanner.nextLine();
	int[] freq = new int[str.length()];    
        char minChar = str.charAt(0), maxChar = str.charAt(0);    
        int i, j, min, max;            
        char[] string = str.toCharArray();         
        for (i = 0; i < string.length; i++) {    
            freq[i] = 1;    
            for (j = i + 1; j < string.length; j++) {    
                if (string[i] == string[j] && string[i] != ' ' && string[i] != '0') {    
                    freq[i]++;    
                    string[j] = '0';    
                }    
            }    
        }    
        min = max = freq[0];    
        for (i = 0; i < freq.length; i++) {    
            if (min > freq[i] && freq[i] != '0') {    
                min = freq[i];    
                minChar = string[i];    
            }    
            if (max < freq[i]) {    
                max = freq[i];    
                maxChar = string[i];    
            }    
        }    
        System.out.println("Minimum occurring character: " + minChar);    
        System.out.println("Maximum occurring character: " + maxChar);
        scanner.close();    
    }
}

class PalindromeChecker {
    public static void main(String[] args) {

        System.out.print("Enter a string: ");
        String originalString = scanner.nextLine();
        String cleanedString = originalString.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        String reversedString = new StringBuilder(cleanedString).reverse().toString();
        if (cleanedString.equals(reversedString)) {

// Java code to Maintain a Bank Account
import java.io.*;

class BankAccount {
    private String depositorName;
    private String accountNumber;
    private String accountType;
    private double balance;
    public BankAccount(String depositorName, String accountNumber, String accountType, double initialBalance) {
        this.depositorName = depositorName;
        this.accountNumber = accountNumber;
        this.accountType = accountType;
        this.balance = initialBalance;
   }
    public void deposit(double amount) {
        if (amount > 0) {
            balance += amount;
            System.out.println("Deposited: " + amount);
        } else {
            System.out.println("Deposit amount must be positive.");
        }
    }
    public void withdraw(double amount) {
        if (amount > 0 && amount <= balance) {
            balance -= amount;
            System.out.println("Withdrew: " + amount);
        } else if (amount > balance) {
            System.out.println("Insufficient balance.");
        } else {
            System.out.println("Withdrawal amount must be positive.");
        }
    }
    public void displayAccountInfo() {
        System.out.println("Depositor Name: " + depositorName);
        System.out.println("Balance: Rs" + balance);
    }

    }
    public String getAccountNumber() {
        return accountNumber;
    }

    }

    public void setAccountNumber(String accountNumber) {

    public void setAccountType(String accountType) {

//Java Programme to implement voting.
import java.util.Random;
import java.util.Scanner;
class Ballot {
    public static void countVotes(int[] ballots) {
        int[] count = new int[5];
        int spoiledBallots = 0;
        for (int ballot : ballots) {
            if (ballot >= 1 && ballot <= 5) {
                count[ballot - 1]++;
            } else {
                spoiledBallots++;
            }
        }
        System.out.println("Vote counts for each candidate:");
        for (int i = 0; i < count.length; i++) {
            System.out.println("Candidate " + (i + 1) + ": " + count[i] + " votes");
        }
        System.out.println("Spoiled ballots: " + spoiledBallots);
    }
    public static void main(String [] args) {
        Scanner scanner = new Scanner(System.in);
        Random random = new Random();
        System.out.print("Enter the number of ballots:");
        int n = scanner.nextInt();
        int[] ballots = new int[n];
        System.out.println("Randomly generating ballot numbers...");
        for (int i = 1; i < n; i++) {
            ballots[i] = random.nextInt(7) + 1;
        }
        countVotes(ballots);
        scanner.close();
    }
}
