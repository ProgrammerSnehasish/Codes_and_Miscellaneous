def ispalindrome(number):
    sum=0
    temp=number
    while number>0:
        d=number%10
        sum=sum*10+d
        number=number//10
    if sum==temp:
        return f"The given Number {temp} is Palindrome number."
    else:
        return f"The given Number {temp} is not Palindrome number."#Function to check palindrome number.

def factorial(number):
  if number==1 or number==0:
    return 1
  elif number<0:
    return "Invalid"
  else:
    return number*factorial(number-1)#Function to print factorial of number

def tribonacci_rec(nterms):
    if nterms==0 or nterms==1 or nterms==2:
        return 0
    elif nterms==3:
        return 1
    else:
        return (tribonacci_rec(nterms - 1)+tribonacci_rec(nterms - 2)+tribonacci_rec(nterms - 3)) #driver code for tribonacci
def print_tribonacci(nterms):
    for i in range(1,nterms):
        print(tribonacci_rec(i)," ",end=" ")#function to print Tribonacci Series
def area_of_circle(radius):
    area_of_the_circle= 22 / 7 * (radius ** 2)
    return area_of_the_circle #function to find area_of_Circle
def circumference_of_circle(radius):
    circumference_of_the_circle= 2 * (22 / 7) * radius
    return circumference_of_the_circle #function to find Circumference_of_Circle
def area_of_triangle(height,base):

def isleapyear(year):
    if year%4==0 and year%100!=0 or year%400==0:
        return f"The given year {year} is leap year."
    else:
        return f"The given year {year} is not leap year."
def convert_celcius_to_farenheit(celcius):
    farenheit=(9*celcius+160)/5
    return farenheit #function to convert temperature from celcius to faranheit
def convert_faranheit_to_celcius(farenheit):

def determine_charactertype(character):
    if ('a' <= character <= 'z')or('A' <= character <= 'Z'):
        return f"The Given Character {character} is an Alphabet."
    elif '0' <= character <= '9':
        return f"The Given Character {character} is a Digit."
    else:
        return f"The Given Character {character} is a Special Character."#Funtion to determine character type.
def isvowel(alphabet):
    if alphabet=='A' or alphabet=='a' or alphabet=='E' or alphabet=='e' or alphabet=='I' or alphabet=='i' or alphabet=='O' or alphabet=='o' or alphabet=='U' or alphabet=='u':
        return f"The given alphabet {alphabet} is a Vowel."
    else:
        return f"The given character is {alphabet} is a Consonant"#function to determine Vowel or Consonant.
def remove_vowel(string):
    vowels=['a', 'e', 'i', 'o', 'u']
    result=[letter for letter in string if letter.lower() not in vowels]
    result=''.join(result)
    return result#function to remove vowel from string.
def string_reverse(string):
    reversed_string=string[::-1]
    return reversed_string#function to reverse string.
