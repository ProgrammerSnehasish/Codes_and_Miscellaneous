#Applications of Numpy
import numpy as np
list=[]
list2=[]
no_of_element=int(input("Enter no of elements in the list:"))
print("Enter data for 1st list:")
for i in range(0,no_of_element):
    element=int(input(f"Enter element {i+1}:"))
    list.insert(i,element)
print("Enter data for 2nd list:")
for i in range(0,no_of_element):
    ele=int(input(f"Enter element {i+1}:"))
    list2.insert(i,ele)
Array=np.array(list)
print(f"The array(of list 1) using numpy is: {Array}")#1-D Numpy Array
D2Array=np.array([list,list2])
print(f"The 2D array using Numpy is:\n {D2Array}")#2-D Numpy Array
print(f"Shape of 2D array: {str(D2Array.shape)}")#Printing Shape of 2D Array
print(f"Datatype of the 2D Array is: {D2Array.dtype}")#Printing Datatype of 2D array
#Multiplication of elements with respect to a scalar using Numpy(Previous operation done in line 14)
scalar_value=int(input("Enter the scalar value with which the list elements are to be multiplied:"))
print(f"The result of the multiplication of list 1 elements w.r.t scalar value is:{Array*scalar_value}")
#Addition,Subtraction,Multiplication,Division,Square Root of list/arrays(using Numpy).
Array1=np.array(list)
Array2=np.array(list2)
print(f"Addition of two Arrays is: {np.add(Array1,Array2)}")#Result of Addition
print(f"Subtraction of two Arrays is: {np.subtract(Array2,Array1)}")#Result of Subtraction
print(f"Multiplication of two Arrays is: {np.multiply(Array1,Array2)}")#Result of Multipy
print(f"Division of two Arrays is: {np.divide(Array1,Array2)}")#Result of Divison
print(f"Square root value of Array1 is: {np.sqrt(Array1)}")#Result of Square Root
#Dot Product of two 2D arrays.
D2Array1=np.array([list,list2])
D2Array2=np.array([list2,list])
print(f"The dot product of two 2D arrays:\n {np.dot(D2Array1,D2Array2)}")#Result of the dot product of two 2D Arrays
#Transpose of 2D Array.
print(f"The Transpose of 2D Array 1:\n {np.transpose(D2Array)}")#Printing Transposed 2D Array.

Row=int(input("Enter Size of rows of matrix:"))
Column=int(input("Enter Size of columns of matrix:"))
ArrayOnes=np.ones((Row,Column))
print(f"The array of ones of dimension {Row}x{Column} is:\n {ArrayOnes}")#Printing Array of Ones
print("Creating Identity Matrices of N dimension:")
Dimension=int(input("Enter Size of matrix:"))
IdentityArray=np.eye(Dimension)
print(f"The Identity array of dimension {Dimension} is:\n {IdentityArray}")#Printing Identity Array

Row=int(input("Enter Size of rows of matrix:"))
Column=int(input("Enter Size of columns of matrix:"))
RandomelementArray=np.random.random((Row,Column))

StartingVal=int(input("Enter Starting Value:"))
